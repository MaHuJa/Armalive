The server. and sessionid are implicit, and not part of the command handed to the extension.

CREATE OR REPLACE FUNCTION server.inf_killed1(
	sessionid integer,
	"when" numeric,
	victim_uid text,  -- getplayeruid
	victim_position text, -- str position
	victim_class text,  -- typeof
	victim_side text, -- side (at start, so no sideEnemy)
	killer_uid text,
	killer_position text,
	killer_class text,
	killer_side text,
	killer_weapon text,	-- weapon
	isTK text	-- not teamkill empty_friendly civilian 
)

CREATE OR REPLACE FUNCTION server.drowned1(
	sessionid integer,
	"when" numeric,
	victim_uid text,
	victim_position text,
	victim_class text,
	victim_side text
)

CREATE OR REPLACE FUNCTION server.died1(
	sessionid integer,
	"when" numeric,
	victim_uid text,
	victim_position text,
	victim_class text,
	victim_side text
)
